document.addEventListener("DOMContentLoaded", function() {

      var durationInput = document.getElementById("duration");

      durationInput.focus();

    });

    function startTimer() {

      var duration = document.getElementById("duration").value; // Get the duration in hours from the input field

      var countdown = duration * 60 * 60; // Convert hours to seconds


      var timer = setInterval(function() {

        var hours = Math.floor(countdown / 3600);

        var minutes = Math.floor((countdown % 3600) / 60);

        var seconds = countdown % 60;


        document.getElementById("timer").innerHTML = hours + "h " + minutes + "m " + seconds + "s ";


        if (countdown <= 0) {

          clearInterval(timer);

          document.getElementById("timer").innerHTML = "Timer completed!";

        }


        countdown--;

      }, 1000);

    }

  </script>
